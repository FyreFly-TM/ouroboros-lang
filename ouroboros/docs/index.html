Getting Started

Installation

Build the Ouroboros compiler from source:

git clone https://github.com/ouroboros-lang/ouroboros.git
cd ouroboros

# On Windows with MinGW
make

# Or compile manually
gcc -Wall -Wextra -std=c99 -o ouroc.exe *.c -lgdi32 -lm

Hello World

Create your first Ouroboros program:

// hello.ouro
function main() {
    print("Hello, Ouroboros!");
    return 0;
}

Compile and run:

ouroc.exe hello.ouro
Output:
Hello, Ouroboros!

Type System

Basic Types

Ouroboros supports both typed and untyped variable declarations:

// Typed declarations
int age = 25;
float pi = 3.14159;
string name = "Ouroboros";
bool is_active = true;

// Untyped declarations (dynamic)
let value = 42;
let message = "Hello";

// Type inference
let count = 0;        // Inferred as int
let ratio = 0.5;      // Inferred as float

Arrays

// Array declaration (planned feature)
int[] numbers = [1, 2, 3, 4, 5];
string[] names = ["Alice", "Bob", "Charlie"];

// Array access
let first = numbers[0];
let last = numbers[4];

Type Casting

// String conversion
let num_str = to_string(42);
let pi_str = to_string(3.14);

// String concatenation
let result = "The answer is " + to_string(42);

Functions

Function Declaration

Functions can be typed or untyped:

// Untyped function
function greet(name) {
    print("Hello, " + name + "!");
}

// Typed function with return type
int add(int a, int b) {
    return a + b;
}

// Void function
void display_message(string msg) {
    print("[INFO] " + msg);
}

// Function with multiple parameters
float calculate_area(float width, float height) {
    return width * height;
}

Function Calls

function main() {
    greet("Alice");
    
    int sum = add(5, 3);
    print("5 + 3 = " + to_string(sum));
    
    float area = calculate_area(10.5, 20.0);
    print("Area: " + to_string(area));
    
    display_message("Program completed");
    
    return 0;
}

Recursion

Functions support recursion:

int factorial(int n) {
    if (n <= 1) {
        return 1;
    }
    return n * factorial(n - 1);
}

int fibonacci(int n) {
    if (n <= 1) {
        return n;
    }
    return fibonacci(n - 1) + fibonacci(n - 2);
}

Scope

Variables have block scope:

int global_var = 100;  // Global scope

function test_scope() {
    int function_var = 200;  // Function scope
    
    if (true) {
        int block_var = 300;  // Block scope
        print(to_string(global_var));    // ✓ Accessible
        print(to_string(function_var));  // ✓ Accessible
        print(to_string(block_var));     // ✓ Accessible
    }
    
    print(to_string(function_var));  // ✓ Accessible
    // print(to_string(block_var));  // ✗ Not accessible
}

Classes & Object-Oriented Programming

Class Declaration

class Person {
    string name;
    int age;
    
    // Constructor
    function Person(string n, int a) {
        this.name = n;
        this.age = a;
    }
    
    // Method
    void introduce() {
        print("Hi, I'm " + this.name + " and I'm " + to_string(this.age) + " years old.");
    }
    
    // Getter
    int get_age() {
        return this.age;
    }
    
    // Setter
    void set_age(int new_age) {
        if (new_age >= 0) {
            this.age = new_age;
        }
    }
}

Creating Objects

function main() {
    // Create new instance
    let person = new Person("Alice", 30);
    
    // Call methods
    person.introduce();
    
    // Access properties
    print("Name: " + person.name);
    print("Age: " + to_string(person.get_age()));
    
    // Modify properties
    person.set_age(31);
    person.introduce();
    
    return 0;
}

Inheritance

class Student extends Person {
    string school;
    float gpa;
    
    function Student(string n, int a, string s, float g) {
        // Call parent constructor
        super(n, a);
        this.school = s;
        this.gpa = g;
    }
    
    void study() {
        print(this.name + " is studying at " + this.school);
    }
    
    // Override parent method
    void introduce() {
        super.introduce();
        print("I study at " + this.school + " with a GPA of " + to_string(this.gpa));
    }
}

Modules & Imports

Creating a Module

Create a file math_utils.ouro:

// math_utils.ouro
int square(int x) {
    return x * x;
}

int cube(int x) {
    return x * x * x;
}

bool is_even(int n) {
    return n % 2 == 0;
}

bool is_prime(int n) {
    if (n <= 1) {
        return false;
    }
    
    int i = 2;
    while (i * i <= n) {
        if (n % i == 0) {
            return false;
        }
        i = i + 1;
    }
    return true;
}

float distance(float x1, float y1, float x2, float y2) {
    float dx = x2 - x1;
    float dy = y2 - y1;
    return sqrt(dx * dx + dy * dy);
}

Using Modules

// main.ouro
import "math_utils";

function main() {
    int num = 7;
    
    print("Square of " + to_string(num) + " = " + to_string(square(num)));
    print("Cube of " + to_string(num) + " = " + to_string(cube(num)));
    
    if (is_even(num)) {
        print(to_string(num) + " is even");
    } else {
        print(to_string(num) + " is odd");
    }
    
    if (is_prime(num)) {
        print(to_string(num) + " is prime");
    }
    
    return 0;
}

Graphics Programming

OpenGL Support

Ouroboros has extensive built-in OpenGL support:

function create_opengl_window() {
    // Initialize OpenGL
    opengl_init();
    
    // Create a window context
    opengl_create_context(800, 600, "My OpenGL Window");
    
    // Create a simple shader
    int shader = opengl_create_shader(
        "vertex_shader_source",
        "fragment_shader_source"
    );
    
    // Main render loop
    while (opengl_is_context_valid()) {
        // Clear the screen
        opengl_clear(0.2, 0.3, 0.3, 1.0);
        
        // Use shader program
        opengl_use_shader(shader);
        
        // Draw geometry
        opengl_draw_arrays(4, 0, 3);  // GL_TRIANGLES
        
        // Swap buffers
        opengl_swap_buffers();
    }
    
    // Cleanup
    opengl_destroy_context();
}

function create_voxel_world() {
    // Create voxel engine
    voxel_engine_create();
    
    // Create a world
    voxel_create_world("MyWorld", 256, 128, 256);
    
    // Generate terrain
    voxel_generate_terrain("perlin", 0.02, 8);
    
    // Set camera position
    voxel_set_camera(128.0, 80.0, 128.0, 0.0, 0.0);
    
    // Enable physics
    voxel_enable_physics();

        // Modify blocks

            voxel_set_block(target_x, target_y, target_z, block_type);
        }
        
        // Render frame
        voxel_render_frame();
        
        // Get performance stats

    // Save world
    voxel_save_world("MyWorld", "saves/world1.vxl");
}

Object-Oriented Game Entity

// game_entity.ouro
class Entity {
    string name;
    float x;
    float y;
    int health;
    
    function Entity(string n, float px, float py) {
        this.name = n;
        this.x = px;
        this.y = py;
        this.health = 100;
    }
    
    void move(float dx, float dy) {
        this.x = this.x + dx;
        this.y = this.y + dy;
    }
    
    void take_damage(int damage) {
        this.health = this.health - damage;
        if (this.health < 0) {
            this.health = 0;
        }
    }
    
    bool is_alive() {
        return this.health > 0;
    }
    
    void display() {
        print(this.name + " at (" + to_string(this.x) + ", " + to_string(this.y) + 
              ") - Health: " + to_string(this.health));
    }
}

class Player extends Entity {
    int score;
    int level;
    
    function Player(string n) {
        super(n, 0.0, 0.0);
        this.score = 0;
        this.level = 1;
    }
    
    void collect_item(int points) {
        this.score = this.score + points;
        print(this.name + " collected item! Score: " + to_string(this.score));
        
        // Level up every 100 points
        if (this.score >= this.level * 100) {
            this.level = this.level + 1;
            print("LEVEL UP! Now level " + to_string(this.level));
        }
    }
}

function main() {
    // Create player
    let player = new Player("Hero");
    player.display();
    
    // Game simulation
    player.move(10.0, 5.0);
    player.collect_item(50);
    player.take_damage(20);
    player.display();
    
    player.move(5.0, -3.0);
    player.collect_item(60);
    player.display();
    
    return 0;
}

Graphics Demo

// graphics_demo.ouro
function draw_rotating_triangle() {
    opengl_init();
    opengl_create_context(800, 600, "Rotating Triangle");
    
    // Create shader
    int shader = opengl_create_shader(
        "#version 330 core\n" +
        "layout (location = 0) in vec3 aPos;\n" +
        "uniform float rotation;\n" +
        "void main() {\n" +
        "    float c = cos(rotation);\n" +
        "    float s = sin(rotation);\n" +
        "    vec3 pos = vec3(aPos.x * c - aPos.y * s, aPos.x * s + aPos.y * c, aPos.z);\n" +
        "    gl_Position = vec4(pos, 1.0);\n" +
        "}",
        "#version 330 core\n" +
        "out vec4 FragColor;\n" +
        "void main() {\n" +
        "    FragColor = vec4(1.0, 0.5, 0.2, 1.0);\n" +
        "}"
    );
    
    // Triangle vertices
    float vertices[] = {
        -0.5, -0.5, 0.0,
         0.5, -0.5, 0.0,
         0.0,  0.5, 0.0
    };
    
    // Create vertex buffer
    int vbo = opengl_create_buffer();
    opengl_bind_buffer(0x8892, vbo); // GL_ARRAY_BUFFER
    opengl_buffer_data(0x8892, sizeof(vertices), vertices, 0x88E4); // GL_STATIC_DRAW
    
    float rotation = 0.0;
    
    // Render loop
    while (opengl_is_context_valid()) {
        // Clear
        opengl_clear(0.2, 0.3, 0.3, 1.0);
        
        // Use shader and set uniform
        opengl_use_shader(shader);
        opengl_set_uniform_float(shader, "rotation", rotation);
        
        // Draw
        opengl_draw_arrays(4, 0, 3); // GL_TRIANGLES
        
        // Update rotation
        rotation = rotation + 0.01;
        
        // Swap buffers
        opengl_swap_buffers();
    }
    
    opengl_destroy_context();
}

function main() {
    draw_rotating_triangle();
    return 0;
}

9. Default-Value Semantics

Fields that you do not explicitly initialise receive safe defaults at compile-time: numbers → 0, floats → 0.0, bool → false, vectors → zero-vector.
The snippet below shows the values printed without manual initialisation.

10. Game-Style Lifecycle

The VM automatically calls Awake, Start, Update… methods if they exist.

public class Mover {
    float x;
    void Awake()  { print("Mover created"); }
    void Start()  { print("Mover starts at " + to_string(x)); }
    void Update() { x = x + 0.1; }
}

When you run the program the engine will:

  1. Create a Mover instance
  2. Call Awake immediately
  3. Call Start once before the frame loop
  4. Invoke Update every frame
